async function verifyProduct() { const urlParams = new URLSearchParams(window.location.search); const code = urlParams.get("code"); if (!code) { document.getElementById("result").innerHTML = "

❌ Invalid QR Code.

"; return; } try { // Append the timestamp to the URL for cache-busting const url = `https://verifiedproductlk.github.io/verifiedproductlk/data.json?t=${new Date().getTime()}`; const response = await fetch(url, { headers: { "Cache-Control": "no-cache" } // Ensures fresh data }); if (!response.ok) { throw new Error("Failed to fetch data"); } const data = await response.json(); console.log("Fetched data:", data); // Debugging // Find the product using the encrypted hash (QR code) const found = data.products.find(item => item.encrypted_hash === code); if (found) { document.getElementById("result").innerHTML = `

✅ Verified Product

Timestamp: ${found.timestamp}

Random Number: ${found.random_number}

QR Code:

`; } else { document.getElementById("result").innerHTML = `

❌ Unverified Product

This product is NOT found in our database.

`; } } catch (error) { console.error("Verification error:", error); document.getElementById("result").innerHTML = "

⚠️ Error verifying product. Please try again later.

"; } }
